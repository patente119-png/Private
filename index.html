import express from "express"; import cors from "cors"; import dotenv from "dotenv"; import OpenAI from "openai"; // .env 파일에서 OPENAI_API_KEY를 불러옵니다. dotenv.config(); const app = express(); const port = process.env.PORT || 3000; app.use(cors()); app.use(express.json({ limit: "1mb" })); app.use(express.static("public")); const client = new OpenAI({ apiKey: process.env.OPENAI_API_KEY, }); function extractJson(text) { try { return JSON.parse(text); } catch (_) { const match = text.match(/\{[\s\S]*\}/); if (!match) throw new Error("AI 응답에서 JSON을 찾지 못했습니다."); return JSON.parse(match[0]); } } app.post("/api/recipe", async (req, res) => { try { const { ingredients } = req.body; if (!ingredients || typeof ingredients !== "string" || ingredients.trim().length < 1) { return res.status(400).json({ error: "재료를 입력해주세요." }); } if (!process.env.OPENAI_API_KEY) { return res.status(500).json({ error: ".env 파일에 OPENAI_API_KEY가 없습니다." }); } const prompt = ` 너는 한국 집밥 레시피 추천 전문가다. 사용자가 입력한 냉장고 재료를 최대한 활용해서 만들 수 있는 집밥 메뉴를 추천해라. 사용자 재료: ${ingredients} 규칙: - 한국어로 답한다. - 현실적인 집밥 메뉴만 추천한다. - 재료가 부족하면 "있으면 좋은 재료"에 적는다. - 사용자가 입력한 재료를 최대한 우선 사용한다. - 너무 비싼 재료나 전문 장비는 피한다. - 결과는 반드시 JSON 하나로만 답한다. 설명 문장은 JSON 밖에 쓰지 않는다. JSON 형식: { "recipes": [ { "menuName": "메뉴명", "category": "반찬/찌개/도시락/한그릇/국/간식 중 하나", "time": "예: 15분", "difficulty": "쉬움/보통/어려움 중 하나", "mainIngredients": ["재료1", "재료2"], "optionalIngredients": ["있으면 좋은 재료1", "있으면 좋은 재료2"], "steps": ["1단계", "2단계", "3단계", "4단계", "5단계"], "tip": "요리 팁 한 문장", "lunchboxTip": "도시락으로 쌀 때의 팁 한 문장" } ] } 메뉴는 4개 추천해라. 각 조리법은 5~8단계로 자세히 작성해라. `; const response = await client.chat.completions.create({ model: "gpt-4o-mini", messages: [ { role: "system", content: "너는 한국 집밥 메뉴와 냉장고 파먹기 레시피를 잘 추천하는 요리 도우미다." }, { role: "user", content: prompt } ], temperature: 0.7, response_format: { type: "json_object" } }); const content = response.choices?.[0]?.message?.content || "{}"; const data = extractJson(content); return res.json(data); } catch (error) { console.error(error); return res.status(500).json({ error: "AI 레시피 생성 중 오류가 발생했습니다.", detail: error.message }); } }); app.listen(port, () => { console.log(`냉장고 AI 집밥 추천기가 실행 중입니다: http://localhost:${port}`); });
